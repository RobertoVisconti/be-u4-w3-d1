Esercizio 1 :

SELECT * FROM fatture WHERE iva = 20


Esercizio 2: 

SELECT * FROM prodotti WHERE EXTRACT (YEAR FROM data_attivazione) = 2017
AND (in_produzione = true OR in_commercio = true)

Esercizio 3:

SELECT * FROM fatture
JOIN clienti ON fatture.id_cliente = clienti.numero_cliente
WHERE importo < 1000;


Esercizio 4:

SELECT numero_fattura, importo, iva, data_fattura, fornitori.denominazione
FROM fatture
JOIN fornitori ON fatture.numero_fornitore = fornitori.numero_fornitore

Esercizio 5:

SELECT EXTRACT(YEAR FROM data_fattura) AS anno,
COUNT(*) AS totale_fatture
FROM fatture 
WHERE iva = 20
GROUP BY EXTRACT(YEAR FROM data_fattura)

Esercizio 6:

SELECT EXTRACT(YEAR FROM data_fattura) AS anno_fatturazione,
COUNT(*) conteggio_fatture,
SUM(importo) AS somma_importi
FROM fatture 
GROUP BY EXTRACT(YEAR FROM data_fattura)
ORDER BY anno_fatturazione ASC


EXTRA 1:

SELECT EXTRACT(YEAR FROM data_fattura) AS anno_tipologia,
COUNT(*)
FROM fatture 
WHERE tipologia = 'A'
GROUP BY EXTRACT(YEAR FROM data_fattura)
HAVING COUNT(*)>2
ORDER BY anno_tipologia ASC


EXTRA 2:

SELECT clienti.regione_residenza,
SUM(importo) AS somma_importi
FROM fatture 
JOIN clienti ON fatture.id_cliente = clienti.numero_cliente
GROUP BY clienti.regione_residenza
ORDER BY somma_importi DESC


EXTRA 3:

SELECT numero_cliente, nome, cognome, anno_di_nascita, importo,
COUNT(numero_fattura) AS totale_fatture
FROM clienti
JOIN fatture ON numero_cliente = id_cliente
WHERE anno_di_nascita = 1980
AND importo > 50
GROUP BY numero_cliente, nome, cognome, anno_di_nascita, importo
ORDER BY totale_fatture DESC
